    Doz.component('doz-text', {
        template(){
            return `
                <div>
                    <input type="text" d-bind="message" placeholder="${this.props.placeholder}"/>
                    <p>Value: ${this.props.message}</p>
                    <button onclick="this.setNew()">Set to hello</button>
                </div>
            `
        },
        props: {
            message: 'ciao'
        },
        setNew() {
            this.props.message = 'hello'
        }
    });

    Doz.component('doz-radio', {
        template(){
            return `
                <div>
                    <input type="radio" d-bind="gender" name="gender" value="male"> Male
                    <input type="radio" d-bind="gender" name="gender" value="female"> Female
                    <input type="radio" d-bind="gender" name="gender" value="other"> Other
                    <p>Value: ${this.props.gender}</p>
                    <button onclick="this.setNew()">Set to female</button>
                </div>
            `
        },
        props: {
            gender: 'other'
        },
        setNew() {
            this.props.gender = 'female'
        }
    });

    Doz.component('doz-select', {
        template(){
            return `
                <div>
                    <select d-bind="item">
                        <option value="" disabled="true" selected="true" hidden="true">Please Choose...</option>
                        <option value="Milk">Milk</option>
                        <option value="Tomato">Tomato</option>
                        <option value="Chicken">Chicken</option>
                    </select>
                    <p>Value: ${this.props.item}</p>
                    <button onclick="this.setNew()">Set to Chicken</button>
                </div>
            `
        },
        props: {
            item: 'Tomato'
        },
        setNew() {
            this.props.item = 'Chicken'
        }
    });

    Doz.component('doz-select-multiple', {
        template(){
            return `
                <div>
                    <select d-bind="item" multiple="multiple">
                        <option value="" disabled="true" selected="true" hidden="true">Please Choose...</option>
                        <option value="Milk">Milk</option>
                        <option value="Tomato">Tomato</option>
                        <option value="Chicken">Chicken</option>
                    </select>
                    <p>Value: ${this.props.item}</p>
                    <button onclick="this.setNew()">Set to Chicken, Milk</button>
                </div>
            `
        },
        props: {
            item: 'Tomato'
        },
        setNew() {
            this.props.item = ['Chicken', 'Milk']
        }
    });

    Doz.component('doz-checkbox', {
        template(){
            return `
                <div>
                    <input type="checkbox" d-bind="done" name="done"> ${this.props.done} <br/><br/>
                    <button onclick="this.setNew()">Set to true</button>
                </div>
            `
        },
        props: {
            done: false
        },
        setNew() {
            this.props.done = true
        }
    });

    Doz.component('doz-checkbox-multiple', {
        template(){
            return `
                <div>
                    <input type="checkbox" d-bind="sauce" name="sauce" value="Mustard"> Mustard
                    <input type="checkbox" d-bind="sauce" name="sauce" value="Mayonnaise"> Mayonnaise
                    <input type="checkbox" d-bind="sauce" name="sauce" value="Ketchup"> Ketchup
                    <input type="checkbox" d-bind="sauce" name="sauce" value="Pink Sauce"> Pink Sauce
                    <p>Value: ${this.props.sauce}</p>
                    <button onclick="this.setNew()">Set to Mustard, Pink Sauce</button>
                </div>
            `
        },
        props: {
            sauce: ['Ketchup', 'Mayonnaise']
        },
        setNew() {
            this.props.sauce = ['Mustard', 'Pink Sauce']
        }
    });
