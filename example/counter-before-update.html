    Doz.component('doz-button-counter', {
        template(){
            return `<div><button d-ref="button" onclick="this.myClickHandler()">${this.props.title}</button> <span class="counter">${this.props.counter}</span></div>`
        },
        props: {
            counter: 0
        },
        myClickHandler() {
            this.props.counter += 1;
        },
        onBeforeUpdate() {
            return this.props.counter < 10;
        },
        onUpdate() {
            if (this.props.counter === 10) {
                this.ref.button.setAttribute('disabled', 'false');
            }
        }
    });

    new Doz({
        root: document.getElementById('app'),
        template: `
            <p>Stop after 10</p>
            <doz-button-counter title="Click me!"></doz-button-counter>
            <doz-button-counter title="Touch me!"></doz-button-counter>
        `
    });
