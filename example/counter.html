HTML

    
<div id="app">
    <doz-button-counter title="Click me!"></doz-button-counter>
    <doz-button-counter title="Touch me!"></doz-button-counter>
</div>

JavaScript

    
    Doz.component('doz-button-counter', {
        template: `<div><button on-click="myClickHandler">{{title}}</button> <span>{{counter}}</span></div>`,
        context: {
            counter: 0,
            myClickHandler() {
                this.counter += 1;
            }
        }
    });

    new Doz({
        el: '#app'
    })
