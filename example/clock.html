    Doz.component('doz-clock', {
        template(){
            return `<div>${this.props.title}: <strong>${this.props.time}</strong></div>`
        },
        props: {
            time: '--:--:--',
        },
        onMount() {
            setInterval(() => this.currentTime(), 1000)
        },
        currentTime() {
            this.props.time = new Date().toLocaleTimeString()
        }
    });

    new Doz({
        root: document.getElementById('app'),
        template: `<doz-clock title="it's"></doz-clock>`
    });
