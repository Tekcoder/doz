<div id="app">
    <doz-clock title="it's"></doz-clock>
</div>

JavaScript

    
    Doz.Component('doz-clock', {
        template: `<div>{{title}}: {{time}}</div>`,
        context: {
            time: '--:--:--',
            onRender() {
                setInterval(() => this.currentTime(), 1000)
            },

            currentTime() {
                this.time = new Date().toLocaleTimeString()
            }
        }
    });

    new Doz({
        el: '#app'
    })
